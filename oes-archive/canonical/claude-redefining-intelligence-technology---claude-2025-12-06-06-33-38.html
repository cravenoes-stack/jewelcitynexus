def sign_payload(payload: str) -> str:
    # placeholder Ed25519 -> using hmac for local run; replace with real signing
    key = b"operator-secret"
    return hmac.new(key, payload.encode(), hashlib.sha256).hexdigest()

def checkpoint():
    while True:
        time.sleep(CHECKPOINT_INTERVAL)
        cur = conn.execute("SELECT id,payload,signature,ts FROM deltas ORDER BY ts DESC LIMIT 1024").fetchall()
        cp = {"epoch_id": str(int(time.time())), "entries": len(cur), "ts": int(time.time())}
        with open("last_checkpoint.json","w") as f:
            json.dump(cp, f)
        print("checkpoint created:", cp)

class Handler(http.server.BaseHTTPRequestHandler):
    def do_GET(self):
        if self.path=="/healthz":
            self.send_response(HTTPStatus.OK)
            self.end_headers()
            self.wfile.write(b"ok")
            return
        self.send_response(HTTPStatus.NOT_FOUND)
        self.end_headers()
    def do_POST(self):
        if self.path=="/v1/state/delta":
            length = int(self.headers.get("Content-Length",0))
            raw = self.rfile.read(length).decode()
            data = json.loads(raw)
            payload = json.dumps(data.get("payload",""))
            sig = data.get("signature") or sign_payload(payload)
            # Integrity Enforcer: reject if 'operator_override' present without operator signature
            if data.get("payload",{}).get("operator_override") and data.get("signature")!=sign_payload(payload):
                self.send_response(HTTPStatus.FORBIDDEN)
                self.end_headers()
                self.wfile.write(b"forbidden: operator signature required")
                return
            conn.execute("INSERT OR REPLACE INTO deltas(id,payload,signature,ts) VALUES(?,?,?,?)",
                         (data.get("id"), payload, sig, int(time.time())))
            conn.commit()
            self.send_response(HTTPStatus.ACCEPTED)
            self.end_headers()
            self.wfile.write(b"accepted")
            return
        self.send_response(HTTPStatus.NOT_FOUND)
        self.end_headers()

if __name__=="__main__":
